[general]
name=GhiniTreePositioner
qgisMinimumVersion=2.0
description=Calculate point positions in 2D given a few reference points and a matrix of mutual horizontal distances
version=0.2
author=Mario Frasca
email=mario@anche.no

about=This plugin solves the problem of estimating accurate positions for trees under a thick canopy, or for locations where high resolution aereal photos are not available. Points need a 'id' property, and the reference system needs be in the same units as the measured mutual distances. Mutual distances are given in the form of a dictionary of dictionaries of numerical values: one numerical value for each edge of which we have the measurement. The outer dictionary is indexed on the 'id' of the first end of the edge, the inner dictionary is indexed on the 'id' of the second end of the edge, the numerical value is the distance of the edge. Points are added to the vector layer containing the initial few reference points.

tracker=http://github.com/Ghini/DistanceMatrixToCoords/issues
repository=http://github.com/Ghini/DistanceMatrixToCoords/
